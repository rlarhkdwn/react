board 테이블 생성

1. root 접속
    mysql -uroot -p1234
2. DB 생성
    create database react;
3. user 생성
    use mysql;
    create user 'reactUser'@'localhost' identified by 'mysql';
    select user from user; => 유저 확인 구문
4. 권한 부여
    grant all privileges on react.* to 'reactUser'@'localhost';
    flush privileges;
5. reactUser 접속
    exit;
    mysql -ureactUser -pmysql
6. DB 선택
    use react;
7. 테이블 생성
    create table board(
        id int not null auto_increment,
        title varchar(100),
        contents varchar(500),
        writer varchar(50),
        reg_date timestamp default current_timestamp,
        primary key(id)
    );
